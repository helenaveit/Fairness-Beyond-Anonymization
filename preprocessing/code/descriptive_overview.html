shorten_prefix <- function(x) {
  # Normalize NBSP to regular space
  x <- gsub("\u00A0", " ", x, fixed = TRUE)

  str_replace(
    x,
    regex(
      paste0(
        "^(",
          # 1) Fachliches & funktionales Kompetenz(-|–|—)Profil:
          "Fachliches\\s*&\\s*funktionales\\s*Kompetenz[-–—]Profil\\s*:\\s*",

          # 3) Sprachkompetenz <LANG> (with flexible spaces, optional colon)
          "Sprachkompetenz\\s+:?\\s*",
        ")"
      ),
      ignore_case = FALSE
    ),
    ""
  )
}

plot_top_for_variable <- function(df_long, var, top_n = 15) {
  stopifnot(all(c("ID","variable","category","subcategory") %in% names(df_long)))
  
  df_long %>%
    filter(variable == var) %>%
    mutate(label = mk_label(category, subcategory)) %>%
    count(label, sort = TRUE, name = "n") %>%
    slice_head(n = top_n) %>%
    arrange(n) %>%
    ggplot(aes(x = reorder(label, n), y = n)) +
    geom_col() +
    coord_flip() +
